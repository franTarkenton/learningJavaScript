    function getGoogleMapTypeConfigs() {
        var types, streetMaps, hybrid;
        types = [];
        streetMaps = {
            label: 'Google Streets',
            params: {
                numZoomLevels: 23
            }};

            label: 'Google Hybrid',
            params: {
                type : google.maps.MapTypeId.HYBRID,
                numZoomLevels : 23,
                MAX_ZOOM_LEVEL : 22
            }

            label: 'Google Satellite', 
            params: {
                type : google.maps.MapTypeId.SATELLITE,

                MAX_ZOOM_LEVEL : 22
            }
        }
